## Problem

1. **`install.sh` es ambiguo** — un solo script que auto-detecta qué binario instalar crea confusión. Un operador de RPi no debería ver lógica de attester y viceversa.

2. **Build logic atrapada en el YAML** — si un build falla en CI no podés reproducirlo localmente porque los pasos son inline en el workflow. Necesitás los mismos scripts corriendo tanto en CI como localmente.

## What to Build

### 1. Crear `scripts/build-device.sh`

Script que cross-compila `device` para ARMv7 musl. Llamado por CI y por `just build-device-release`.

```bash
#!/usr/bin/env bash
# build-device.sh — Cross-compile `device` for ARMv7 musl (Raspberry Pi).
#
# Prerequisites: Rust target armv7-unknown-linux-musleabihf, cargo-zigbuild, zig
# Required env: RELEASE_VERSION (e.g. "v0.1.0" or "dev")
# Output: device-${RELEASE_VERSION}-armv7-unknown-linux-musleabihf + .sha256

set -euo pipefail

TARGET="armv7-unknown-linux-musleabihf"

if [ -z "${RELEASE_VERSION:-}" ]; then
  echo "ERROR: RELEASE_VERSION is not set." >&2
  echo "       Usage: RELEASE_VERSION=v0.1.0 bash scripts/build-device.sh" >&2
  exit 1
fi

if ! command -v zig >/dev/null 2>&1; then
  echo "ERROR: 'zig' not found. Install Zig: https://ziglang.org/download/" >&2
  exit 1
fi

if ! cargo zigbuild --version >/dev/null 2>&1; then
  echo "ERROR: 'cargo-zigbuild' not found. Install: cargo install cargo-zigbuild" >&2
  exit 1
fi

if ! rustup target list --installed | grep -q "^${TARGET}$"; then
  echo "ERROR: Rust target '${TARGET}' not installed." >&2
  echo "       Run: rustup target add ${TARGET}" >&2
  exit 1
fi

echo "==> Building contracts (ABI required by protocol crate) ..."
cd contracts && forge build && cd ..

echo "==> Cross-compiling device for ${TARGET} ..."
cargo zigbuild --release --package device --target "${TARGET}"

ARTIFACT="device-${RELEASE_VERSION}-${TARGET}"
echo "==> Creating artifact: ${ARTIFACT}"
cp "target/${TARGET}/release/device" "${ARTIFACT}"

echo "==> Generating checksum ..."
if command -v sha256sum >/dev/null 2>&1; then
  sha256sum "${ARTIFACT}" > "${ARTIFACT}.sha256"
elif command -v shasum >/dev/null 2>&1; then
  shasum -a 256 "${ARTIFACT}" > "${ARTIFACT}.sha256"
else
  echo "ERROR: sha256sum/shasum not found." >&2; exit 1
fi

echo ""
echo "Build complete."
echo "  Artifact: ${ARTIFACT}"
echo "  Size:     $(du -sh "${ARTIFACT}" | cut -f1)"
```

### 2. Crear `scripts/build-attester.sh`

Script que compila `attester` para el host actual (x86_64 Linux o macOS arm64). Sin cross-compilation.

```bash
#!/usr/bin/env bash
# build-attester.sh — Build `attester` for the current host platform.
#
# Supported hosts: x86_64-unknown-linux-gnu, aarch64-apple-darwin
# Required env: RELEASE_VERSION (e.g. "v0.1.0" or "dev")
# Output: attester-${RELEASE_VERSION}-${TARGET} + .sha256

set -euo pipefail

if [ -z "${RELEASE_VERSION:-}" ]; then
  echo "ERROR: RELEASE_VERSION is not set." >&2
  echo "       Usage: RELEASE_VERSION=v0.1.0 bash scripts/build-attester.sh" >&2
  exit 1
fi

detect_target() {
  local os arch
  os="$(uname -s)"; arch="$(uname -m)"
  case "${os}" in
    Linux)
      case "${arch}" in
        x86_64) echo "x86_64-unknown-linux-gnu" ;;
        *) echo "ERROR: Unsupported Linux arch: ${arch}. Supported: x86_64." >&2; exit 1 ;;
      esac ;;
    Darwin)
      case "${arch}" in
        arm64) echo "aarch64-apple-darwin" ;;
        *) echo "ERROR: Unsupported macOS arch: ${arch}. Supported: arm64." >&2; exit 1 ;;
      esac ;;
    *) echo "ERROR: Unsupported OS: ${os}." >&2; exit 1 ;;
  esac
}

TARGET="$(detect_target)"

if ! rustup target list --installed | grep -q "^${TARGET}$"; then
  echo "ERROR: Rust target '${TARGET}' not installed." >&2
  echo "       Run: rustup target add ${TARGET}" >&2
  exit 1
fi

echo "==> Building contracts (ABI required by protocol crate) ..."
cd contracts && forge build && cd ..

echo "==> Building attester for ${TARGET} ..."
cargo build --release --package attester --target "${TARGET}"

ARTIFACT="attester-${RELEASE_VERSION}-${TARGET}"
echo "==> Creating artifact: ${ARTIFACT}"
cp "target/${TARGET}/release/attester" "${ARTIFACT}"

echo "==> Generating checksum ..."
if command -v sha256sum >/dev/null 2>&1; then
  sha256sum "${ARTIFACT}" > "${ARTIFACT}.sha256"
elif command -v shasum >/dev/null 2>&1; then
  shasum -a 256 "${ARTIFACT}" > "${ARTIFACT}.sha256"
else
  echo "ERROR: sha256sum/shasum not found." >&2; exit 1
fi

echo ""
echo "Build complete."
echo "  Artifact: ${ARTIFACT}"
echo "  Target:   ${TARGET}"
echo "  Size:     $(du -sh "${ARTIFACT}" | cut -f1)"
```

### 3. Crear `install-device.sh` (reemplaza `install.sh` para RPi)

Solo para Raspberry Pi ARMv7. Falla rápido con mensaje claro en cualquier otra plataforma.

```bash
#!/usr/bin/env bash
# install-device.sh — Install HardTrust `device` on Raspberry Pi (ARMv7 only).
# For Ubuntu/macOS use install-attester.sh instead.
#
# Env: HARDTRUST_VERSION (default: latest), INSTALL_DIR (default: /usr/local/bin)

set -euo pipefail
REPO="OWNER/hardtrust"   # TODO: replace OWNER
INSTALL_DIR="${INSTALL_DIR:-/usr/local/bin}"
BINARY="device"
TARGET="armv7-unknown-linux-musleabihf"

guard_platform() {
  local os arch
  os="$(uname -s)"; arch="$(uname -m)"
  [ "${os}" = "Linux" ] || {
    echo "ERROR: This script is only for Raspberry Pi (Linux ARMv7)." >&2
    echo "       For Ubuntu/macOS use install-attester.sh." >&2; exit 1; }
  case "${arch}" in
    armv7l|armv7) ;;  # OK
    x86_64)
      echo "ERROR: This is a server/desktop (x86_64). Use install-attester.sh." >&2; exit 1 ;;
    aarch64)
      echo "ERROR: ARMv7 32-bit required, detected aarch64. Check https://github.com/${REPO}/issues." >&2; exit 1 ;;
    *)
      echo "ERROR: Unsupported arch: ${arch}." >&2; exit 1 ;;
  esac
}

resolve_version() {
  [ -n "${HARDTRUST_VERSION:-}" ] && { echo "${HARDTRUST_VERSION}"; return; }
  local v
  v=$(curl -fsSL "https://api.github.com/repos/${REPO}/releases/latest" \
    | grep '"tag_name"' | sed 's/.*"tag_name": *"\([^"]*\)".*/\1/')
  [ -n "${v}" ] || { echo "ERROR: Could not resolve latest release." >&2; exit 1; }
  echo "${v}"
}

verify_checksum() {
  local file="$1" expected="$2" actual
  if command -v sha256sum >/dev/null 2>&1; then
    actual=$(sha256sum "${file}" | awk '{print $1}')
  elif command -v shasum >/dev/null 2>&1; then
    actual=$(shasum -a 256 "${file}" | awk '{print $1}')
  else
    echo "WARNING: Skipping checksum (sha256sum/shasum not found)." >&2; return 0
  fi
  [ "${actual}" = "${expected}" ] || {
    echo "ERROR: Checksum mismatch! Expected: ${expected} Got: ${actual}" >&2; exit 1; }
  echo "Checksum OK"
}

main() {
  echo "HardTrust Device Installer (Raspberry Pi ARMv7)"
  echo "================================================"
  guard_platform
  local version artifact base_url
  version="$(resolve_version)"
  artifact="${BINARY}-${version}-${TARGET}"
  base_url="https://github.com/${REPO}/releases/download/${version}"
  echo "Version: ${version} | Target: ${TARGET}"
  local tmp; tmp="$(mktemp -d)"; trap 'rm -rf "${tmp}"' EXIT
  echo "Downloading ${artifact} ..."
  curl -fsSL --progress-bar "${base_url}/${artifact}" -o "${tmp}/${BINARY}"
  local expected
  expected=$(curl -fsSL "${base_url}/${artifact}.sha256" | awk '{print $1}')
  verify_checksum "${tmp}/${BINARY}" "${expected}"
  echo "Installing to ${INSTALL_DIR}/${BINARY} ..."
  if [ -w "${INSTALL_DIR}" ]; then
    cp "${tmp}/${BINARY}" "${INSTALL_DIR}/${BINARY}"; chmod +x "${INSTALL_DIR}/${BINARY}"
  else
    sudo cp "${tmp}/${BINARY}" "${INSTALL_DIR}/${BINARY}"; sudo chmod +x "${INSTALL_DIR}/${BINARY}"
  fi
  echo "Done. Run 'device --help' to get started."
}
main "$@"
```
